      if (key !== 'NaN') { map[key] = 1; }

    var count = 0;
    for (i = 0; i < (predicted || []).length; i++) {
      key = String(parseInt(predicted[i], 10));
      if (key !== 'NaN' && map[key]) { count++; }
    }
    return count;
  }

  function forensicDateKey(v){
    var raw=String(v||'').replace(/\s+/g,' ').replace(/^\s+|\s+$/g,'');
    if(!raw){return '';}
    var t=Date.parse(raw);
    if(!isNaN(t)){var d=new Date(t); var m=d.getMonth()+1; var day=d.getDate(); return d.getFullYear()+'-'+(m<10?'0'+m:String(m))+'-'+(day<10?'0'+day:String(day));}
    var mt=raw.match(/(\d{4})[-\/](\d{1,2})[-\/](\d{1,2})/);
    if(mt){return mt[1]+'-'+(parseInt(mt[2],10)<10?'0'+parseInt(mt[2],10):String(parseInt(mt[2],10)))+'-'+(parseInt(mt[3],10)<10?'0'+parseInt(mt[3],10):String(parseInt(mt[3],10)));}
    return raw.toLowerCase();
  }
  function forensicSameSet(a,b){
    var aa=(a||[]).map(String).filter(Boolean).sort();
    var bb=(b||[]).map(String).filter(Boolean).sort();
    if(!aa.length || !bb.length || aa.length!==bb.length){return false;}
    for(var i=0;i<aa.length;i++){if(aa[i]!==bb[i]){return false;}}
    return true;
  }
  function forensicRowMatchesDraw(r, draw){
    if(!r || !draw){return false;}
    var dateOk=!!(forensicDateKey(draw.date) && forensicDateKey(r.date) && forensicDateKey(draw.date)===forensicDateKey(r.date));
    var mainOk=forensicSameSet(r.actualMain, draw.main);
    var extraOk=(!draw.extra.length && !r.actualExtra.length) || forensicSameSet(r.actualExtra, draw.extra);
    if(mainOk && extraOk){return true;}
    if(dateOk && mainOk){return true;}
    return false;
  }
  function forensicHasBatchMarker(r){
    if(!r){return false;}
    if(r.batchUuid){return true;}
    if((r.batchTotal||0)>=2 || (r.batchIndex||0)>0){return true;}
    if(r.sourceBatchId && r.sourceBatchId !== '0'){return true;}
    if(r.sourceBatchRunId && r.sourceBatchRunId !== '0'){return true;}
    if(/\b(batch|9-run|nine run|precision|mle_precision)\b/i.test(String((r.method||'')+' '+(r.label||'')+' '+(r.sourceBatchType||'')))){return true;}
    return false;
  }
  function forensicBatchKey(r){
    var method = String(r.analysisMethodKey || r.method || '');
    var target = String(r.predictionTargetId || r.predictionTargetKey || '');
    if (r.sourceBatchId && r.sourceBatchId !== '0') { return 'sourcebatch|' + r.sourceBatchId + '|' + target + '|' + method; }
    if (r.batchUuid) { return 'batch|' + r.batchUuid + '|' + target + '|' + method; }
    if ((r.batchTotal || 0) >= 2) { return 'batchtotal|' + String(r.batchTotal || 0) + '|' + target + '|' + method + '|' + forensicDateKey(r.date || ''); }
    return 'unbatched|' + target + '|' + method + '|' + forensicDateKey(r.date || '');
  }
  function forensicBestGroup(rows){
    var groups = {}, order = [], bestKey = '', bestLen = -1, bestScore = -1, bestRecent = -1;
    rows.forEach(function(r){
      var k = forensicBatchKey(r);
      if (!groups[k]) { groups[k] = []; order.push(k); }
      groups[k].push(r);
    });
    order.forEach(function(k){
      var g = groups[k], score = 0, recent = 0;
      g.forEach(function(r){ score += (r.totalHits || 0) + ((r.top20 || 0) * 0.01) + ((r.top10 || 0) * 0.02); recent = Math.max(recent, Date.parse(r.date || '') || 0); });
      var lenScore = Math.min(g.length, 9);
      if (lenScore > bestLen || (lenScore === bestLen && recent > bestRecent) || (lenScore === bestLen && recent === bestRecent && score > bestScore)) { bestKey = k; bestLen = lenScore; bestScore = score; bestRecent = recent; }
    });
    return groups[bestKey] || [];
  }
  function latestBatch(rows, draw){
    if (!rows.length || !draw) { return []; }
    var exact = rows.filter(function(r){ return forensicRowMatchesDraw(r, draw); });
    if (!exact.length) { return []; }
    var batchRows = exact.filter(forensicHasBatchMarker);
    if (!batchRows.length) { return []; }
    var grouped = forensicBestGroup(batchRows);
    grouped.sort(function(a, b){
      var ai = a.sourceBatchRunOrder || a.batchIndex || 0;
      var bi = b.sourceBatchRunOrder || b.batchIndex || 0;
      if (ai !== bi) { return ai - bi; }
      return String(b.date).localeCompare(String(a.date));
    });
    if (grouped.length > 9) { grouped = grouped.slice(0, 9); }
    return grouped;
  }

  function bestRun(rows){
    var best = null;
    rows.forEach(function(r){
      if (!best) { best = r; return; }
      if ((r.totalHits || 0) > (best.totalHits || 0)) { best = r; return; }
      if ((r.totalHits || 0) === (best.totalHits || 0) && (r.top20 || 0) > (best.top20 || 0)) { best = r; return; }
      if ((r.totalHits || 0) === (best.totalHits || 0) && (r.top20 || 0) === (best.top20 || 0) && r.isBest && !best.isBest) { best = r; }
    });
    return best;
  }
  function savedRecommendationRankForRow(row){
    if (!row) { return 0; }
    if ((row.savedRecommendationRankRaw || 0) > 0) { return row.savedRecommendationRankRaw; }
    if ((row.sourceBatchRunOrder || 0) > 0) { return row.sourceBatchRunOrder; }
    if ((row.batchIndex || 0) > 0) { return row.batchIndex; }
    if ((row.visibleRunNumber || 0) > 0) { return row.visibleRunNumber; }
    if ((row.printedRowOrder || 0) > 0) { return row.printedRowOrder; }
    return 0;
  }
  function assignSavedRecommendationRanks(rows){
    var used = {};
    var fallback = 1;
    (rows || []).forEach(function(row){
      var proposed = savedRecommendationRankForRow(row);
      if (proposed > 0 && !used[proposed]) {
        row.savedRecommendationRank = proposed;
        used[proposed] = 1;
        return;
      }
      while (used[fallback]) { fallback++; }
      row.savedRecommendationRank = fallback;
      used[fallback] = 1;
      fallback++;
    });
  }
  function recommendationBySavedRank(rows, rank){
    var i;
    for (i = 0; i < (rows || []).length; i++) {
      if ((rows[i].savedRecommendationRank || 0) === rank) { return rows[i]; }
    }
    return null;
  }

  // [[MLE_PHASE2D_CHUNKED_VISUAL_SCORING_RUNNER_2026_05_23]]
  // The legacy visual-scoring repair pass used to process every lottery card in one
  // synchronous sweep. On large MyLotto pages this created long tasks and hurt LCP.
  // Keep the same repairs, but process cards in small chunks so the browser can paint
  // and respond between batches.
  var __mleV139RunQueued = false;
  function processOneCard(card){
    if (!card) { return; }
    ensureBottomToggle(card);
    enforceGuidance(card);
    enforceCanonicalCreate(card);
    enforceRange(card);
    enforceReport(card);
    syncButtons(card, isOpen(card));
  }
  function processCardChunk(cards, index){
    var i = index || 0;
    var started = (window.performance && performance.now) ? performance.now() : Date.now();
    var budgetMs = 24;
    var maxPerChunk = 4;
    var processed = 0;
    while (i < cards.length && processed < maxPerChunk) {
      processOneCard(cards[i]);
      i++;
      processed++;
      if (((window.performance && performance.now) ? performance.now() : Date.now()) - started > budgetMs) { break; }
    }
    if (i < cards.length) {
      window.setTimeout(function(){ processCardChunk(cards, i); }, 35);
    }
  }
  function runAll(){
    if (__mleV139RunQueued) { return; }
    __mleV139RunQueued = true;
    window.setTimeout(function(){
      __mleV139RunQueued = false;
      processCardChunk(qsa('.mle-advisory-card[data-lottery-id]'), 0);
    }, 0);
  }

    var owner = closest(bottomToggle, '.mle-advisory-card');
    if (!owner) { return; }

  }, true);

        reviewPendingButtonsFound: hasReview,
        skaiButtonsFound: hasSkai,
        collapseButtonsFound: hasCollapse,
        missingFeatureWarnings: warnings.join(' | ')
      });
    });
    if (window.console && console.table) { console.table(rows); }
    else if (window.console) { console.log(rows); }
    return rows;
  };

  // Phase 2D: one queued startup pass is enough after the asset is loaded.
  // Additional passes are triggered by user interaction and MutationObserver-style
  // repairs elsewhere; avoid repeated heavy startup sweeps.
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function(){ setTimeout(runAll, 300); });
  } else {
    setTimeout(runAll, 300);
  }
  window.addEventListener('load', function(){ setTimeout(runAll, 1200); });
